- saturation
- smooth
- zoom

        {
            "Name": "Zoom",
            "Action": {
                "Offset": "30"
        },
        {
            "Name": "Saturation",
            "Action": {
                "Offset": "-10"
        },
        {
            "Name": "Smooth",
            "Action": {
                "Offset": "10"
        }
    ]

{{$input}}
